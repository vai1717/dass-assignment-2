\begin{enumerate}[label=Test Case \arabic*]
  \item \textbf{Player cannot buy property if balance is insufficient}
    \begin{itemize}
      \item \textit{Why:} Ensures the buy logic checks player balance and prevents negative balances.
      \item \textit{Covers:} Branch in property purchase logic.
      \item \textit{Edge:} Player with zero or low balance.
      \item \textit{Result:} [Pass/Fail].
    \end{itemize}
  \item \textbf{Player sent to jail after 3 consecutive doubles}
    \begin{itemize}
      \item \textit{Why:} Checks the jail rule is enforced after three doubles.
      \item \textit{Covers:} Branch in dice roll logic.
      \item \textit{Edge:} Player rolls doubles three times.
      \item \textit{Result:} [Pass/Fail].
    \end{itemize}
  \item \textbf{Cannot pay rent on mortgaged property}
    \begin{itemize}

      \item \textit{Covers:} Branch in rent payment logic.

      \item \textit{Result:} [Pass/Fail].

  \item \textbf{Passing Go awards salary}
    \begin{itemize}
